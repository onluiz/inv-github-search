        <v-btn

          large block
          @click="newNote">

        <note-card :notes="notes"></note-card>

    },
  },
  data() {
    return {
      notes: [],

  computed: {
    userSearch() {

    },
  },
  mounted() {
    this.notes = notesLib.getNotes();
  },
